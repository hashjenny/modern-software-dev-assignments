```csharp
using System.Net;
using System.Text.RegularExpressions;
using notehub_blazor.Models;

namespace notehub_blazor.Services;

public static class SearchHelpers
{
    public static IReadOnlyList<string> Tokenize(string query)
    {
        if (string.IsNullOrWhiteSpace(query)) return Array.Empty<string>();

        return query
            .Split(' ', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries)
            .Select(token => token.ToLowerInvariant())
            .Distinct()
            .ToList();
    }

    public static int Score(Note note, IReadOnlyList<string> tokens)
    {
        var title = note.Title.ToLowerInvariant();
        var content = note.Content.ToLowerInvariant();
        var score = 0;

        foreach (var token in tokens)
        {
            if (title.Contains(token)) score += 2;
            if (content.Contains(token)) score += 1;
        }

        return score;
    }

    public static string BuildSnippet(string content, IReadOnlyList<string> tokens, int length = 120)
    {
        if (string.IsNullOrWhiteSpace(content)) return string.Empty;
        if (tokens.Count == 0) return Truncate(content, length);

        var lower = content.ToLowerInvariant();
        var index = tokens
            .Select(token => lower.IndexOf(token, StringComparison.OrdinalIgnoreCase))
            .Where(i => i >= 0)
            .DefaultIfEmpty(0)
            .Min();

        var start = Math.Max(0, index - length / 4);
        var slice = content.Substring(start, Math.Min(length, content.Length - start));

        if (start > 0) slice = "..." + slice;
        if (start + length < content.Length) slice += "...";

        return slice;
    }

    public static string Highlight(string text, IReadOnlyList<string> tokens)
    {
        var encoded = WebUtility.HtmlEncode(text);
        if (tokens.Count == 0) return encoded;

        var pattern = string.Join("|", tokens.Select(Regex.Escape));
        return Regex.Replace(encoded, pattern, "<mark>$0</mark>", RegexOptions.IgnoreCase);
    }

    private static string Truncate(string text, int length)
    {
        if (text.Length <= length) return text;
        return text.Substring(0, length) + "...";
    }
}
```

```csharp
public async Task<IEnumerable<Note>> SearchAsync(string query)
{
    var tokens = SearchHelpers.Tokenize(query);
    if (tokens.Count == 0) return Array.Empty<Note>();

    IQueryable<Note> queryable = _context.Notes;
    foreach (var token in tokens)
    {
        var pattern = $"%{token}%";
        queryable = queryable.Where(n =>
            EF.Functions.Like(n.Title.ToLower(), pattern) ||
            EF.Functions.Like(n.Content.ToLower(), pattern));
    }

    var results = await queryable.ToListAsync();

    return results
        .Select(note => new { Note = note, Score = SearchHelpers.Score(note, tokens) })
        .OrderByDescending(item => item.Score)
        .ThenByDescending(item => item.Note.UpdatedAt)
        .Select(item => item.Note)
        .ToList();
}
```

- [ ] **Step 2: Update SearchResults to compute tokens, snippets, highlights**

```razor
@using Microsoft.AspNetCore.Components
@using notehub_blazor.Services

@code {
    private string query = string.Empty;
    private string searchQuery = string.Empty;
    private List<Note> notes = new();
    private Dictionary<int, string> categoryNames = new();
    private IReadOnlyList<string> tokens = Array.Empty<string>();

    protected override async Task OnInitializedAsync()
    {
        var uri = Navigation.ToAbsoluteUri(Navigation.Uri);
        var queryParams = QueryHelpers.ParseQuery(uri.Query);
        query = queryParams.TryGetValue("q", out var qValue) ? qValue.ToString() : string.Empty;
        searchQuery = query;

        if (!string.IsNullOrEmpty(query))
        {
            await SearchAsync();
        }
    }

    private async Task SearchAsync()
    {
        tokens = SearchHelpers.Tokenize(query);
        if (tokens.Count == 0)
        {
            notes.Clear();
            return;
        }

        notes = (await NoteService.SearchAsync(query)).ToList();

        var categories = await CategoryService.GetAllAsync();
        foreach (var category in categories)
        {
            categoryNames[category.Id] = category.Name;
        }
    }

    private MarkupString? GetHighlightedTitle(Note note)
    {
        if (tokens.Count == 0) return null;
        return new MarkupString(SearchHelpers.Highlight(note.Title, tokens));
    }

    private MarkupString? GetHighlightedExcerpt(Note note)
    {
        if (tokens.Count == 0) return null;
        var snippet = SearchHelpers.BuildSnippet(note.Content, tokens);
        return new MarkupString(SearchHelpers.Highlight(snippet, tokens));
    }
}
```
